def tri_shell(T):
    N = len(T)
    p = 0
    
    while p < N:
        p = 3 * p + 1
    
    while p != 0:
        p = p // 3
        
        for i in range(p, N):
            v = T[i]
            j = i
            
            while j >= p and T[j - p] > v:
                T[j] = T[j - p]
                j -= p
            T[j] = v

# Exemple d'utilisation
T = [35, 33, 42, 10, 14, 19, 27, 44, 26, 31]
tri_shell(T)
print(T)  # [10, 14, 19, 26, 27, 31, 33, 35, 42, 44]
